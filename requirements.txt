# Зависимости приложения «Mark» (Система маркировки)
#
# ВАЖНО: используется именно PyQt5 (НЕ PyQt6).
# Это требование целевого оборудования: планшетный терминал на Intel
# Celeron J1900 не поддерживает AVX, и PyQt6 там не запускается.
# Не заменяйте PyQt5 на PyQt6.
#
# Установка:  python -m pip install -r requirements.txt

PyQt5            # графический интерфейс (Qt5)
pyserial         # связь с весами по COM-порту (RS-232 / RS-485)
openpyxl         # чтение Excel: листы «продукт», «изготовил», «цех»
pywin32          # печать на термопринтер (модуль win32print)
tzdata           # база часовых поясов для zoneinfo на Windows

# Только для сборки .exe (для запуска из исходников не требуется):
# pyinstaller
